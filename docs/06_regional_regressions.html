Model specification

The first regional consistency model is:

log(BF-NEE + 1) ~ log(natural ecosystem loss + 1) + log(weight sum + 1)

where weight_sum represents the aggregated commodity-consumption and production-attribution component.

Regional regression models

Show code 
fit_region <- function(x) {
  lm(
    log_bfnee ~ log_natural_loss + log_weight_sum,
    data = x
  )
}

regional_models <- df_model %>%
  group_by(continent) %>%
  group_modify(~ {
    model <- fit_region(.x)
    
    tidy_model <- tidy(model)
    glance_model <- glance(model)
    
    tidy_model %>%
      mutate(
        r_squared = glance_model$r.squared,
        adj_r_squared = glance_model$adj.r.squared,
        n_obs = glance_model$nobs
      )
  }) %>%
  ungroup()

regional_models

Coefficient comparison

Show code 
regional_coefficients <- regional_models %>%
  filter(term != "(Intercept)")

ggplot(
  regional_coefficients,
  aes(
    x = continent,
    y = estimate
  )
) +
  geom_point(size = 2) +
  geom_errorbar(
    aes(
      ymin = estimate - 1.96 * std.error,
      ymax = estimate + 1.96 * std.error
    ),
    width = 0.15
  ) +
  facet_wrap(~ term, scales = "free_y") +
  labs(
    title = "Regional regression coefficients",
    subtitle = "Error bars show approximate 95% confidence intervals",
    x = "Region",
    y = "Coefficient estimate"
  ) +
  theme_minimal() +
  theme(
    axis.text.x = element_text(
      angle = 45,
      hjust = 1
    )
  )
