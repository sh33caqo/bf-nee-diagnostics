valid <- read_csv(

    dir_diag,

  show_col_types = FALSE
)

valid_iso3 <- valid %>%
  filter(valid_for_analysis) %>%
  pull(iso3)

Overview

This page summarizes global BF-NEE trends based on countries that passed all processing and scale diagnostics.

Show code 
global_trend <- diag %>%
  filter(
    iso3 %in% valid_iso3,
    status == "ok",
    suspicious_old_scale == FALSE
  ) %>%
  group_by(year, transition) %>%
  summarise(
    global_bfnee_km2 =
      sum(bfnee_sum_km2, na.rm = TRUE),

    global_natural_loss_km2 =
      sum(natural_loss_sum_km2, na.rm = TRUE),

    n_countries =
      n_distinct(iso3),

    .groups = "drop"
  ) %>%
  arrange(year) %>%
  mutate(
    bfnee_index =
      global_bfnee_km2 /
      first(global_bfnee_km2) * 100,

    natural_loss_index =
      global_natural_loss_km2 /
      first(global_natural_loss_km2) * 100,

    bfnee_share_of_loss =
      global_bfnee_km2 /
      global_natural_loss_km2
  )
