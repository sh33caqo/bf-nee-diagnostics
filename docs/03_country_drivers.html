library(dplyr)
library(readr)
library(tidyr)
library(ggplot2)
library(scales)
library(DT)

dir_diag <- "~/Symobio/Data/Processed/BF_NEE_Diagnostics"

diag <- read_csv(
  file.path(dir_diag, "bfnee_country_diagnostics_all_transitions.csv"),
  show_col_types = FALSE
)

valid <- read_csv(
  file.path(dir_diag, "bfnee_valid_countries.csv"),
  show_col_types = FALSE
)

valid_iso3 <- valid %>%
  filter(valid_for_analysis) %>%
  pull(iso3)

Country contributions to global BF-NEE

Show code 
country_trend <- diag %>%
  filter(
    iso3 %in% valid_iso3,
    status == "ok",
    suspicious_old_scale == FALSE
  ) %>%
  group_by(iso3, year, transition) %>%
  summarise(
    bfnee_km2 = sum(bfnee_sum_km2, na.rm = TRUE),
    natural_loss_km2 = sum(natural_loss_sum_km2, na.rm = TRUE),
    weight_sum = mean(weight_sum, na.rm = TRUE),
    .groups = "drop"
  )

global_by_year <- country_trend %>%
  group_by(year, transition) %>%
  summarise(
    global_bfnee_km2 = sum(bfnee_km2, na.rm = TRUE),
    .groups = "drop"
  )

country_trend <- country_trend %>%
  left_join(global_by_year, by = c("year", "transition")) %>%
  mutate(
    share_global_bfnee = bfnee_km2 / global_bfnee_km2
  )

Top countries by period

Show code 
top_countries_by_period <- country_trend %>%
  group_by(year, transition) %>%
  slice_max(bfnee_km2, n = 15, with_ties = FALSE) %>%
  ungroup()

ggplot(
  top_countries_by_period,
  aes(
    x = reorder(iso3, bfnee_km2),
    y = bfnee_km2
  )
) +
  geom_col() +
  coord_flip() +
  facet_wrap(~ transition, scales = "free_y") +
  scale_y_continuous(labels = comma) +
  labs(
    title = "Top BF-NEE countries by transition",
    subtitle = "Validated countries only",
    x = "Country",
    y = "BF-NEE (km²)"
  ) +
  theme_minimal()
